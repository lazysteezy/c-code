#include <stdio.h>
#include <string.h>

int main(){
    char string1[20];
    int i, length;
    int temp = 0;

    printf("Enter a string:");
    scanf("%s", string1);



    length=0;
    while(string1[length]!=NULL)
    {
        length++;
    }

    for(i=0;i < length ;i++){
        if(string1[i] != string1[length-i-1]){
            temp = 1;
            break;
   }
}

    if (temp) {
        printf("%s is not a palindrome", string1);
    }
    else {
        printf("%s is a palindrome", string1);
    }
    return 0;
}
